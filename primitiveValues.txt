package AssignmentDayOne;
import java.lang.String;

public class PrimitiveValuesDisplay {
    public static void main(String[] args) {
        byte a = 1;
        int b = 2;
        float c = 3.0F;
        double d = 4.0000;
        String e = "Hello";
        char f = 'a';
        boolean flag = true;

        System.out.println("The Value of a is "+ a + "\n" + "The value of b is " + b );
        System.out.println("The Value of c is "+ c + "\n" + "The value of d is " + d );
        System.out.println("The Value of a is "+ e + "\n" + "The value of f is " + f );
        System.out.println("The value of flag is " + flag);
    }
}
